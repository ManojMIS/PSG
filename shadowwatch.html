🧹 Step 1 — Missing Value Imputation

Different imputation strategies are applied per-column based on data type, distribution shape, and domain context. Numeric environmental signals use KNN imputation; economic indicators use forward-fill with regional median fallback.

import pandas as pd from sklearn.impute import KNNImputer, SimpleImputer # --- Strategy 1: KNN Imputation for satellite/environmental fields --- env_cols = ['forest_cover_pct', 'ndvi_score', 'deforestation_delta'] knn_imp = KNNImputer(n_neighbors=5, weights='distance') df[env_cols] = knn_imp.fit_transform(df[env_cols]) # --- Strategy 2: Median imputation for economic fields (robust to outliers) --- econ_cols = ['gdp_growth_pct', 'trade_openness', 'corruption_index'] med_imp = SimpleImputer(strategy='median') df[econ_cols] = med_imp.fit_transform(df[econ_cols]) # --- Strategy 3: Forward-fill for time-series trade data --- trade_cols = ['gold_export_tons', 'timber_export_usd'] df[trade_cols] = df.sort_values('date')[trade_cols].ffill().bfill() # --- Sentinel value handling: replace -99 with NaN then re-impute --- df['vessel_speed_knots'].replace(-99.0, pd.NA, inplace=True) df['vessel_speed_knots'].fillna(df['vessel_speed_knots'].median(), inplace=True) print(f"Missing values after imputation: {df.isnull().sum().sum()}") # → 0
🔍 Step 2 — Outlier Detection & Removal

Trade and export fields have extreme outliers caused by data entry errors, country aggregation mismatches, and genuine anomalies. We use IQR capping for training data only — genuine anomalies in inference data are retained as signals.

import numpy as np def iqr_cap(series, multiplier=1.5): """Cap outliers at IQR bounds — preserves distribution shape.""" Q1, Q3 = series.quantile([0.25, 0.75]) IQR = Q3 - Q1 lower, upper = Q1 - multiplier * IQR, Q3 + multiplier * IQR return series.clip(lower, upper) def zscore_flag(series, threshold=3.0): """Flag rows where z-score exceeds threshold as anomaly candidates.""" z = (series - series.mean()) / series.std() return (np.abs(z) > threshold).astype(int) # Cap outliers in TRAINING data only numeric_fields = ['gold_export_tons', 'ais_gap_hours', 'dark_vessel_count'] for col in numeric_fields: df_train[col] = iqr_cap(df_train[col]) # Flag outliers in INFERENCE data as potential anomalies for col in numeric_fields: df_infer[f'{col}_outlier_flag'] = zscore_flag(df_infer[col]) # Remove exact duplicates (same region + date + source) df.drop_duplicates(subset=['region_id', 'date', 'source'], inplace=True) print(f"Records after dedup: {len(df)}") # 48,294 → 44,871

📐 Normalization & Scaling Strategy

Different scaling methods are chosen per-field based on distribution shape. Export quantities are log-transformed (right-skewed), NDVI scores are min-max scaled (bounded 0–1), and economic indicators use RobustScaler to resist remaining outliers.

from sklearn.preprocessing import MinMaxScaler, RobustScaler, StandardScaler # --- Log transform for right-skewed export quantities --- log_cols = ['gold_export_tons', 'timber_export_usd', 'news_mention_count'] for col in log_cols: df[f'{col}_log'] = np.log1p(df[col]) # log1p handles zeros safely # --- MinMax [0,1] for bounded satellite indices --- minmax_cols = ['ndvi_score', 'forest_cover_pct', 'corruption_index'] mm_scaler = MinMaxScaler() df[minmax_cols] = mm_scaler.fit_transform(df[minmax_cols]) # --- RobustScaler for economic fields (median-centered, IQR-scaled) --- robust_cols = ['gdp_growth_pct', 'trade_openness'] rob_scaler = RobustScaler(quantile_range=(25, 75)) df[robust_cols] = rob_scaler.fit_transform(df[robust_cols]) # --- StandardScaler for vessel metrics (approx normal distribution) --- std_cols = ['vessel_speed_knots', 'ais_gap_hours', 'road_density_km2'] std_scaler = StandardScaler() df[std_cols] = std_scaler.fit_transform(df[std_cols]) # Save scalers for inference-time inverse transforms import joblib joblib.dump(mm_scaler, 'models/minmax_scaler.pkl') joblib.dump(rob_scaler, 'models/robust_scaler.pkl') joblib.dump(std_scaler, 'models/std_scaler.pkl')

# --- Feature Engineering Pipeline --- def build_features(df): # Z-score of export vs rolling 12-month baseline df['export_zscore'] = df.groupby('region_id')['gold_export_tons_log'].transform( lambda x: (x - x.rolling(12).mean()) / (x.rolling(12).std() + 1e-8) ) # Forest loss rate (monthly delta) df['forest_loss_rate'] = df.groupby('region_id')['forest_cover_pct'].diff(-1) # NDVI trend slope (OLS over 6-month window) from scipy.stats import linregress df['ndvi_trend_slope'] = df.groupby('region_id')['ndvi_score'].transform( lambda x: x.rolling(6).apply(lambda w: linregress(range(6), w).slope) ) # Trade mirror gap (import-export discrepancy) df['trade_mirror_gap'] = ( (df['gold_export_tons'] - df['partner_import_tons']).abs() / ((df['gold_export_tons'] + df['partner_import_tons']) / 2 + 1e-8) ) # AIS dark ratio df['ais_dark_ratio'] = df['ais_gap_hours'] / (df['voyage_total_hours'] + 1e-8) # Corruption × trade interaction term df['corruption_trade_product'] = df['corruption_index'] * df['export_zscore'].clip(0) # Multi-signal count (co-occurrence) thresholds = { 'export_zscore': 2.0, 'forest_loss_rate': -0.05, 'ais_dark_ratio': 0.3, 'nlp_risk_density': 0.1 } df['multi_signal_count'] = sum( (df[col] > thresh).astype(int) for col, thresh in thresholds.items() ) return df

🚨 Anomaly Detection Methods

Three complementary anomaly detection approaches run in parallel. Their outputs are ensembled via weighted majority vote. Statistical methods catch simple outliers; model-based approaches find complex multi-dimensional anomalies.

from sklearn.ensemble import IsolationForest from sklearn.neighbors import LocalOutlierFactor from sklearn.svm import OneClassSVM import numpy as np FEATURE_MATRIX = df_fused[FINAL_FEATURE_COLS].values # --- Method 1: Isolation Forest (primary) --- # Randomly partitions space; anomalies require fewer splits → shorter path iso_forest = IsolationForest( n_estimators=200, contamination=0.08, # expect ~8% anomaly rate max_features=0.7, random_state=42 ) iso_scores = iso_forest.fit_predict(FEATURE_MATRIX) iso_anomaly_score = -iso_forest.score_samples(FEATURE_MATRIX) # higher = more anomalous # --- Method 2: Local Outlier Factor (neighborhood-based) --- # Compares local density of point vs its k-neighbors lof = LocalOutlierFactor(n_neighbors=20, contamination=0.08, novelty=True) lof.fit(FEATURE_MATRIX) lof_scores = -lof.score_samples(FEATURE_MATRIX) # --- Method 3: Z-score per individual feature (interpretable baseline) --- zscore_matrix = np.abs((FEATURE_MATRIX - FEATURE_MATRIX.mean(axis=0)) / (FEATURE_MATRIX.std(axis=0) + 1e-8)) zscore_anomaly = zscore_matrix.max(axis=1) # worst-case z-score per row # --- Ensemble: weighted combination --- weights = {'iso': 0.50, 'lof': 0.30, 'zscore': 0.20} ensemble_score = ( weights['iso'] * (iso_anomaly_score / iso_anomaly_score.max()) + weights['lof'] * (lof_scores / lof_scores.max()) + weights['zscore'] * (zscore_anomaly / zscore_anomaly.max()) ) df_fused['anomaly_ensemble_score'] = ensemble_score

📤 Risk Score Computation & Output Schema

The final risk score for each region is a weighted ensemble of all model outputs, normalized to [0, 1]. Scores are then thresholded into HIGH/MEDIUM/LOW classes and serialized to the dashboard via a FastAPI endpoint.

from scipy.special import expit # sigmoid def compute_final_risk_score(row): """ Compute composite risk score from all model outputs. Returns float in [0, 1]. Thresholds: HIGH>0.7, MED 0.4-0.7, LOW<0.4 """ # Normalized component scores (all in [0,1]) iso = row['iso_anomaly_score_norm'] # Isolation Forest rf = row['rf_risk_proba_high'] # Random Forest P(HIGH) lstm = row['lstm_next_month_proba'] # LSTM forecast nlp = row['nlp_article_risk_score'] # TF-IDF + LogReg multi = row['multi_signal_count'] / 4.0 # co-occurrence (0-4 → 0-1) # Weighted ensemble composite = ( 0.30 * iso + 0.30 * rf + 0.20 * lstm + 0.10 * nlp + 0.10 * multi ) # Apply sigmoid to sharpen decision boundary near thresholds risk_score = float(expit((10 * composite) - 5)) risk_level = "HIGH" if risk_score > 0.7 else ("MEDIUM" if risk_score > 0.4 else "LOW") return { "region_id": row['region_id'], "risk_score": round(risk_score, 3), "risk_level": risk_level, "component_scores": {"iso": iso, "rf": rf, "lstm": lstm, "nlp": nlp}, "top_signals": get_top_signals(row, n=5), "timestamp": pd.Timestamp.utcnow().isoformat() } df_fused['risk_output'] = df_fused.apply(compute_final_risk_score, axis=1) output_json = df_fused['risk_output'].tolist() # → Serialized to /api/v1/risk-zones endpoint (FastAPI)
